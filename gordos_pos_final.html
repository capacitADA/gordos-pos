function printComanda() { if (!suscripcionActiva) return; if(!orden.length) { toast("No hay productos"); return; } let nombre = document.getElementById("c-nombre").value.trim() || "Cliente"; let tipo = document.getElementById("c-tipo").value; let obs = document.getElementById("c-obs").value.trim(); let numStr = String(pedidoNum).padStart(3,"0"); let fecha = new Date().toLocaleString("es-CO", {hour12: false}); let total = getTotal(); let empaque = getEmpaque(); let line = "-----------------------------"; let html = `
GORDO'S BURGER
Calle 34 #25-02 · Bucaramanga
318 487 8966
${line}
Pedido: #${numStr}
Fecha: ${fecha}
Cliente: ${nombre}
Tipo: ${tipo === "llevar" ? "Para llevar" : "En local"}
${obs ? `
Obs: ${obs}
` : ""}
${line}
`; orden.forEach(item => { let sub = item.price * item.qty; html += `
${item.qty}x ${item.name} $${(sub/1000).toFixed(0)}K
`; }); if(empaque > 0) { html += `
Empaque llevar $${(empaque/1000).toFixed(0)}K
`; } html += `
${line}
TOTAL $${(total/1000).toFixed(0)}K
${line}
¡Gracias por preferirnos!
`; let ventana = window.open("", "_blank", "width=300,height=400"); ventana.document.write(` Comanda #${numStr} ${html} `); ventana.document.close(); ventana.print(); }
